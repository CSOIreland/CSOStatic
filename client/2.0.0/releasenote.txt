>> csolib_startSpinner >> cso.spinner.start
>> csolib_stopSpinner >> cso.spinner.stop
>> csolib_loadContent >> cso.content
>> csolib_navContent >> cso.content.navigate
>> csolib_AjaxCallbackFunction >> cso.ajax.callback
>> csolib_AjaxCall >> cso.ajax.request
>> csolib_isURLParamSet >> cso.uri.isParam
>> csolib_parseURL >> cso.uri.parse
>> csolib_getNoHeader >> cso.uri.getNoHeader
>> csolib_getNoNavbar >> cso.uri.getNoNavbar
>> csolib_getNoFooter >> cso.uri.getNoFooter
>> csolib_getBody >> cso.uri.getBody

>> C_CSOLIB_NOHEADER >>  C_CSOLIB_URI_NOHEADER
